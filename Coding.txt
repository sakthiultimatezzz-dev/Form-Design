Answer = DLookup("[Application_no]", "ENTRY", "[Application_No] = '" & Me.Txtapp & "'")
If Not IsNull(Answer) Then
MsgBox "Duplicate App Number Found" & vbCrLf & "Please Search and Update.", vbCritical + vbOKOnly + vbDefaultButton1, "Duplicate"
Cancel = True
Exit Sub
Me.Txtapp.Undo
 
 Else:
 End If


If Not Len(Trim(Txtapp)) = 13 Then
MsgBox "Please Enter the 13 Digit number", vbInformation, "APPLICATION MISMATCH"
Cancel = True
Exit Sub
End If

If Not IsNumeric(Trim(Txtapp)) Then
MsgBox "Please Enter The number only", vbQuestion, "CODE MISMATCH"
Cancel = True
Exit Sub
End If
End Sub

 Else:
 End If
